colors = {
    'Budget': '#2E86AB',
    'Mid-range': '#F6C85F',
    'Luxury': '#D1495B'
}

for category in oyo['price_category'].dropna().unique():
    subset = oyo[oyo['price_category'] == category]

    plt.scatter(
        subset['Price'],
        subset['Rating'],
        label=category,
        color=colors.get(category, 'gray'),
        alpha=0.7,
        edgecolors='black'
    )

plt.title('Price vs Rating by Price Category')
plt.xlabel('Price')
plt.ylabel('Rating')
plt.legend(title='Category')
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()
